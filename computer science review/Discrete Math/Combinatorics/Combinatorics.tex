\question{Permutations and combinations}

\begin{itemize}
\item Permutations (care about order)
\begin{itemize}
\item place n objects in k positions
\item 
\[
\Myperm{k} = \frac{n!}{(n-k)!}\quad
\]
\end{itemize}
\item combinations (dont care about order)
\begin{itemize}
\item divide by n! because compared to permutation n places(order) to place first choice, n-1 to place second...
\item 
\[
\Mycomb{k} = \frac{n!}{k!(n-k)!}={n \choose k}\quad
\]
\end{itemize}
\end{itemize}

\question{Bonomial Identity}

\[
(x+y)^n = \sum_{k=0}^n {n \choose k}x^{n-k}y^k = \sum_{k=0}^n {n \choose k}x^{k}y^{n-k}\quad
\]

\[
{n \choose 0} = {n \choose n} = 1 \quad \text{for all integers }\quad
\]
