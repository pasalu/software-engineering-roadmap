\question{Big O}

\begin{itemize}
  \item space O(n)
  \item time
\begin{itemize}
  \item search worst O(n), average O(1)
  \item insert worst O(n), average O(1)
  \item delete worst O(n), average O(1)
\end{itemize}
\end{itemize}

\question{Advantages}

\begin{itemize}
  \item faster than other structures on large entries
  \item efficient when max entries known (dont have to resize)
\end{itemize}

\question{Disadvantages}

\begin{itemize}
  \item have to resize for more data
\end{itemize}

\question{Uses}

\begin{itemize}
  \item associative arrays (arrays index through arbitrary strings)
  \item database indexing
  \item caches
  \item sets (?)
  \item object rep (key is method or object, value is pointer to member or method)
\end{itemize}

\question{Properties}

\begin{itemize}
  \item keys have to be hash able (able to compute numeric value from it)
  \item entries in no particular order
\end{itemize}

\question{Creating a Hash Table}

\begin{lstlisting}
public HashMap<Integer, Student> buildMap(Student[] students) {
	HashMap<Integer, Student> map = new HashMap<Integer, Student>();
	for (Student s : students) map.put(s.getId(), s);
	return map;
}
\end{lstlisting}

\question{Notes}

\begin{itemize}
  \item alternative to array to implement stack and queue
  \item allows any length
  \item can be made more efficient with better fit hash function
\end{itemize}
